import pandas as pd
from sklearn.preprocessing import OneHotEncoder, StandardScaler
import umap.umap_ as umap
import plotly.express as px

# --- 1. OneHot encode nominal columns ---
nominal_cols = ['Gene_Mutation_Type', 'Gene/Factor', 'Chromosome_Location',
                'Function', 'Effect', 'Category']

ohe = OneHotEncoder(sparse_output=False, handle_unknown='ignore')
nominal_encoded = ohe.fit_transform(df[nominal_cols])

nominal_encoded_df = pd.DataFrame(
    nominal_encoded,
    columns=ohe.get_feature_names_out(nominal_cols),
    index=df.index
)

# --- 2. Merge everything except target ---
X_encoded_df = pd.concat(
    [df.drop(nominal_cols + ['Disease_Stage'], axis=1),
     nominal_encoded_df],
    axis=1
)

y = df['Disease_Stage']   # target for coloring

# --- 3. Scale ALL features together ---
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X_encoded_df)

# --- 4. UMAP in 3D ---
umap_3d = umap.UMAP(n_components=3, random_state=42)
X_umap_3d = umap_3d.fit_transform(X_scaled)

# --- 5. Plot ---
umap_df = pd.DataFrame(X_umap_3d, columns=['UMAP1', 'UMAP2', 'UMAP3'])
umap_df['Disease_Stage'] = y.values

fig = px.scatter_3d(
    umap_df, x='UMAP1', y='UMAP2', z='UMAP3',

import tensorflow as tf
from tensorflow import keras
from tensorflow.keras import layers

num_classes = len(y.unique())

model = keras.Sequential([
    layers.Input(shape=(X_train.shape[1],)),
    layers.Dense(64, activation='relu'),
    layers.Dropout(0.3),
    layers.Dense(32, activation='relu'),
    layers.Dense(num_classes, activation='softmax')   # multi-class output
])

model.compile(
    optimizer='adam',
    loss='sparse_categorical_crossentropy',
    metrics=['accuracy']
)
